create table groups(
int id primary key generated by default as identity ,
name varchar(32)
);
create table student(
int id primary key generated by default as identity ,
name varchar(32) ,
lastname varchar(32) ,
email varchar(32) ,
year int ,
username varchar(32) ,
password varchar(32) ,
role varchar(32) ,
group_id int references group(id) on delete set null ,
code varchar(64) );

create table student(

year int(32));

create table subject(
int id primary key generated by default as identity ,
name varchar(32) ,
description varchar(255) ,
teacher_id int references(teacher_id)) on delete set null);

create table enrolled_students(
student_id int references(student(id)) on delete set null ,
subject_id int references(subject(id)) on delete set null);
